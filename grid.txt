шаблон контейнера по колонкам
grid-template-columns: 300px 200px;

gap: 20px 50px - отступы между колонками (между строками / иежду колонками)

// по умолчанию высота под контент
grid-template-rows: 300px 100px;

 grid-auto-rows: 250px; - по умолчанию для остальных колонок

 grid-auto-flow: column; - имеет значение row, что означает, что переполнение идет вертикально - элементы добавляются в новую строку.
Изменяя grid-auto-flow на column, мы меняем направление перетекания с вертикального на горизонтальное. Элементы начинают добавляться в новую колонку вместо новой строки.


относительные размеры колонок
grid-template-columns: 50% 50%; - не дает правильной ширины
grid-template-columns: 1fr 1fr; - правильное разделение
grid-template-columns: repeat(3, 1fr); /* 1fr 1fr 1fr*/
grid-template-columns: repeat(3, 1fr) auto; /* 1fr 1fr 1fr calc(auto)*/


grid-template-rows: 1fr 1fr; - для строк работает если только задана высота контейнера (таблицы)
grid-template-rows: repeat(3, 1fr); - равномерное распределение высоты строк по высоте

/// изменение ширины элемента
.item_x {
    grid-column: span 2;
    grid-column: span 2;
}

.item_x {
    grid-column: span 4;    
}

.item_x {
    grid-column: span 8; (при темплейте 6)  - автоматическое изменение шаблона
}

2-ой вариант задания ширины колонок через линии
.item11 {
    grid-column-start: 3;
    grid-column-end: 8;
}

.item11 {
    grid-column-start: 1;
    grid-column-end: -1;
}
альтернативная запись
.item11 {
    grid-column: 1 / -1;
}

.item11 {
    grid-column: span 2 / 3; (объединение колонок со 1 - 3 линию)
}


.item11 {
    grid-column: span 2 / 3;
    grid-row:  1 / -1; - работает только если строк больше чем нужно для элементов
}
